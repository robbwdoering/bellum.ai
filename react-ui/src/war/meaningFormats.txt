/**
 * CONDITIONAL OBJECT 
 * This object represents some ongoing rule that will change how a dice roll is conducted.
 */
COND_OBJ: {
	type: [COND_OBJ_TYPE],
	subtype: [COND_OBJ_SUBTYPE],
	params: {} 
}

/**
 * MODIFICATION OBJECT 
 * This object represents some ongoing rule that will change how a dice roll is conducted.
 */
MODS: {
	type: [MODS_OBJ_TYPE],
	target: [SELF|OTHER|ABSTRACT_NAME|UNIT_NAME] // undefined implies self
	subtype: [MOD_OBJ_SUBTYPE],
	expiration: [BattlePhaseIndicator] // undefined implies every phase
	cond: [ COND_OBJ... ]
	params: {} 
}
